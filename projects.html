1. ตรวจเลขคู่-คี่

โค้ดนี้ใช้ while loop ในการตรวจสอบเลขคู่หรือเลขคี่ที่ผู้ใช้ป้อนเข้ามา


int num;
printf("ป้อนตัวเลข (0 เพื่อออก): ");
scanf("%d", &num);
while (num != 0) {
    if (num % 2 == 0) {
        printf("%d เป็นเลขคู่\n", num);
    } else {
        printf("%d เป็นเลขคี่\n", num);
    }
    printf("ป้อนตัวเลขอีกครั้ง: ");
    scanf("%d", &num);
}

2. นับจำนวนตัวอักษร

โค้ดนี้ใช้ while loop เพื่ออ่านข้อความและนับจำนวนตัวอักษรที่ป้อนเข้ามา


char str[100];
int count = 0, i = 0;
printf("ป้อนข้อความ: ");
gets(str);
while (str[i] != '\0') {
    count++;
    i++;
}
printf("ข้อความมี %d ตัวอักษร\n", count);
